class Solution {
    public String reverseLeftWords(String s, int n) {
        int length = s.length();
        char[] ar = new char[n];
        char[] ar1 = new char[length - n];
        int size = 0;
        for(int i = 0; i < length; i++){
            if (i < n){
                ar[size++] = s.charAt(i);
            }else{
                ar1[i-size] = s.charAt(i);
            }
          
        }
        String result = new String(ar, 0, size);
        result = new String(ar1, 0, length-size).concat(result);
        return result;
    }
}
