        stack1 = new Stack<>();
        stack2 = new Stack<>();
    }
    
    public void push(int x) {
        stack1.add(x);
        if(stack2.isEmpty() || stack2.peek() >= x){
            stack2.push(x);
        }
    }
    
    public void pop() {
        if (stack1.peek().equals(stack2.peek())){
            stack2.pop();
        }
        stack1.pop();
    }
    
    public int top() {
        return stack1.peek();
    }
    
    public int min() {
        return stack2.peek();
    }
}

        int length = s.length();
        char[] ar = new char[n];
        char[] ar1 = new char[length - n];
        int size = 0;
        for(int i = 0; i < length; i++){
            if (i < n){
                ar[size++] = s.charAt(i);
            }else{
                ar1[i-size] = s.charAt(i);
            }

        return result;
    }
